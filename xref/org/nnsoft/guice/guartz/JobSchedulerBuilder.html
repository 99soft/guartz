252     /**
253      * Add the produced {@code Job} to the given {@code Scheduler},
254      * and associate the related {@code Trigger} with it.
255      *
256      * Users <b>MUST NOT</b> use this method!
257      *
258      * @param scheduler The given {@code Scheduler}
259      * @throws Exception If any error occurs
260      */
261     @Inject
262     public void schedule( Scheduler scheduler )
263         throws Exception
264     {
265         if ( cronExpression == null && trigger == null )
266         {
267             throw new ProvisionException( format( "Impossible to schedule Job '%s' without cron expression",
268                                                   jobClass.getName() ) );
269         }
270         if ( cronExpression != null && trigger != null )
271         {
272           throw new ProvisionException( format( "Impossible to schedule Job '%s' with cron expression " +
273                                                 "and an associated Trigger at the same time", jobClass.getName() ) );
274         }
275 
276         scheduler.scheduleJob( newJob( jobClass )
277                                .withIdentity( DEFAULT.equals( jobName ) ? jobClass.getName() : jobName, jobGroup )
278                                .requestRecovery( requestRecovery )
279                                .storeDurably( storeDurably ).build(),
280                                ( trigger == null ) ?
281                                newTrigger()
282                                .withIdentity( DEFAULT.equals( triggerName ) ? jobClass.getCanonicalName() : triggerName,
283                                               triggerGroup )
284                                .withSchedule( cronSchedule( cronExpression )
285                                               .inTimeZone( timeZone ) )
286                                               .withPriority( priority )
287                                               .build()
288                                 : trigger);
289 
290     }
